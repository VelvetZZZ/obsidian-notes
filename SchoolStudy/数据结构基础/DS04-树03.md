int Height_of_BST( int preorder[], int N ) {

// Base case: If the tree (or subtree) is empty, return -1.

// This ensures that a single leaf node (N=1) returns max(-1, -1) + 1 = 0.

if ( N == 0 ) {

return -1;

}

int root = preorder[0];

int i;

// Find the split point between left and right subtrees.

// In a BST preorder sequence, the first element larger than the root

// marks the start of the right subtree.

for ( i = 1; i < N; i++ ) {

if ( preorder[i] > root ) {

break;

}

}

// Elements from index 1 to i-1 are the left subtree.

// Count of left nodes = (i - 1)

int left_height = Height_of_BST( preorder + 1, i - 1 );

// Elements from index i to N-1 are the right subtree.

// Count of right nodes = N - i

int right_height = Height_of_BST( preorder + i, N - i );

// Height is the max depth of children + 1 for the root edge

return ( left_height > right_height ? left_height : right_height ) + 1;

}

void PercolateUp( int p, PriorityQueue H )

{

int parent;

ElementType tmp = H->Elements[p];

  

/* 当未到根结点并且当前值小于父节点 */

for ( parent = p / 2; parent > 0 && H->Elements[parent] > tmp; parent /= 2 ) {

H->Elements[p] = H->Elements[parent]; /* 父节点下移 */

p = parent;

}

  

H->Elements[p] = tmp; /* 放置到最终位置 */

}
